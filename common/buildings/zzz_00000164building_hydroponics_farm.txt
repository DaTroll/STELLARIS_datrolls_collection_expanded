building_hydroponics_farm = {
    base_buildtime = @b1_time
    category = resource
    potential = {
        exists = owner
        owner = {
            OR = {
                country_uses_food = yes
                is_ai = no
            }
        }
        NOT = {
            is_planet_class = pc_machine
        }
        NOT = {
            has_modifier = resort_colony
        }
    }
    allow = {
        OR = {
            owner = {
                is_ai = no
            }
            AND = {
                owner = {
                    is_ai = yes
                    has_monthly_income = {
                        resource = food
                        value < 30
                    }
                }
                free_district_slots = 0
            }
        }
        is_occupied_flag = no
    }
    destroy_trigger = {
        exists = owner
        OR = {
            owner = {
                is_ai = yes
                country_uses_food = no
            }
            planet = {
                NOT = {
                    is_planet_class = pc_habitat
                }
                free_building_slots = 0
                owner = {
                    is_ai = yes
                    has_monthly_income = {
                        resource = food
                        value > 100
                    }
                }
            }
            planet = {
                free_building_slots = 0
                owner = {
                    is_ai = yes
                    has_monthly_income = {
                        resource = food
                        value > 250
                    }
                }
            }
        }
    }
    inline_script = {
        script = jobs/farmers_add
        FARMER_AMOUNT = 3
        ANGLER_AMOUNT = 2
    }
    resources = {
        category = planet_buildings
        cost = {
            minerals = @b1_minerals
        }
        upkeep = {
            energy = 2
        }
    }
    prerequisites = {
        "tech_hydroponics"
    }
    upgrades = {
        building_eutab_hydroponics_tower
        building_avb_hydroponics_farm_2
    }
}
